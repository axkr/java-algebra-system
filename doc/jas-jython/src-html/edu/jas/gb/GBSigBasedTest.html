132    /**
133     * Test GBase.
134     */
135    public void testGBase() {
136        L = new ArrayList<GenPolynomial<BigRational>>();
137
138        a = fac.random(kl, ll, el, q);
139        b = fac.random(kl, ll, el, q);
140        c = fac.random(kl, ll, el, q);
141        d = fac.random(kl, ll, el, q);
142        e = d; //fac.random(kl, ll, el, q );
143
144        if (a.isZERO() || b.isZERO() || c.isZERO() || d.isZERO()) {
145            return;
146        }
147
148        assertTrue("not isZERO( a )", !a.isZERO());
149        L.add(a);
150
151        L = bb.GB(L);
152        assertTrue("isGB( { a } )", bb.isGB(L));
153
154        assertTrue("not isZERO( b )", !b.isZERO());
155        L.add(b);
156        //System.out.println("L = " + L.size() );
157
158        L = bb.GB(L);
159        assertTrue("isGB( { a, b } )", bb.isGB(L));
160
161        assertTrue("not isZERO( c )", !c.isZERO());
162        L.add(c);
163
164        L = bb.GB(L);
165        assertTrue("isGB( { a, b, c } )", bb.isGB(L));
166
167        assertTrue("not isZERO( d )", !d.isZERO());
168        L.add(d);
169
170        L = bb.GB(L);
171        assertTrue("isGB( { a, b, c, d } )", bb.isGB(L));
172
173        assertTrue("not isZERO( e )", !e.isZERO());
174        L.add(e);
175
176        L = bb.GB(L);
177        assertTrue("isGB( { a, b, c, d, e } )", bb.isGB(L));
178    }

021    public static void runJythonRegister() {
022        ScriptEngine jas = new ScriptEngineManager().getEngineByExtension("py");
023        System.out.println("JAS service discovered: " + jas);
024        if ( jas == null ) {
025            ScriptEngineManager scriptManager = new ScriptEngineManager();
026            scriptManager.registerEngineExtension("py", new PyScriptEngineFactory());
027            jas = scriptManager.getEngineByExtension("py");
028        }
029        if ( jas == null ) {
030            System.out.println("No JAS engine found");
031            return;
032        }
033        System.out.println("Using JAS engine: " + jas);
034        try {
035            long millis = System.currentTimeMillis();
036            //String ex = "x = 1; print(4*2*x)";
037            String ex = "from jas import PolyRing, ZZ; r = PolyRing(ZZ(),\"x,y,z\",PolyRing.lex); print str(r);"
038                      + "[one,x,y,z] = r.gens(); p = ((x*y)+z)**33; print \"p = \" + str(p)";
039            System.out.println("input:  " + ex);
040            Object ans = jas.eval(ex);
041            millis = System.currentTimeMillis() - millis;
042            System.out.println("answer: " + ans);
043            System.out.println("evaluation took " + millis);
044
045        } catch (ScriptException e) {
046            e.printStackTrace();
047        } catch (Exception e) {
048            e.printStackTrace();
049        }
050    }

080    public static void runJruby() {
081        ScriptEngine jas = new ScriptEngineManager().getEngineByExtension("rb");
082        System.out.println("JAS service discovered: " + jas);
083        if ( jas == null ) {
084            System.out.println("No JAS engine found");
085            return;
086        }
087        System.out.println("Using JAS engine: " + jas);
088        try {
089            long millis = System.currentTimeMillis();
090            //String ex = "x = 1; print(4*2*x)";
091            String ex = "require \"jas\"; r = PolyRing.new(ZZ(),\"x,y,z\",PolyRing.lex); puts r.to_s;"
092                      + "one,x,y,z = r.gens(); p = ((x*y)+z)**33; puts \"p = \" + p.to_s";
093            System.out.println("input:  " + ex);
094            Object ans = jas.eval(ex);
095            millis = System.currentTimeMillis() - millis;
096            System.out.println("answer: " + ans);

098
099        } catch (ScriptException e) {

101        } catch (Exception e) {
